Optimized for GitHub and Azure DevOps pull request comments.

terraform show -json plan.tfplan | tfplan2md [OPTIONS]
INSTALLATION

OPTIONS

--template NAME Use template: default|summary|minimal
--show-sensitive Show sensitive values (default: masked)
--report-title TEXT Custom report title

$ cat examples/github-actions.yml
name: Terraform Plan
on: [pull_request]
jobs:
plan:
runs-on: ubuntu-latest
steps:
- name: Generate report
run: |
terraform show -json plan.tfplan | \
docker run -i oocx/tfplan2md > plan.md
- name: Comment PR
uses: actions/github-script@v6
with:
script: |
const fs = require('fs');
const body = fs.readFileSync('plan.md', 'utf8');
github.rest.issues.createComment({...body});
